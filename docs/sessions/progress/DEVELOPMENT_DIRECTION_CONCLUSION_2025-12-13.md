```python
# scripts/validate_docs.py
"""
Pre-commit hook for document consistency validation.

Checks:
1. Naming convention compliance
2. Completion grade presence (if "Complete" keyword used)
3. SSOT conflict detection
4. Date consistency (document date vs file date)
"""

import re
import sys
from pathlib import Path
from datetime import datetime

def validate_naming(file_path: Path) -> list[str]:
    """Check naming convention compliance."""
    errors = []
    name = file_path.stem

    # Pattern checks
    patterns = {
        r'^WEEK\d+_DAY\d+_[A-Z_]+$': 'Progress document',
        r'^PHASE_[A-Z]_[A-Z_]+_\d{4}-\d{2}-\d{2}$': 'Completion document',
        r'^[A-Z_]+_ANALYSIS_\d{4}-\d{2}-\d{2}$': 'Analysis document',
        r'^[A-Z_]+_GUIDE$': 'Guide document',
    }

    # At least one pattern should match
    if not any(re.match(p, name) for p in patterns):
        errors.append(f"Naming convention violation: {name}")

    return errors

def validate_completion_grade(file_path: Path) -> list[str]:
    """Check completion grade presence."""
    errors = []
    content = file_path.read_text(encoding='utf-8')

    # If "Complete" or "ì™„ë£Œ" in content
    if re.search(r'(Complete|ì™„ë£Œ)', content, re.IGNORECASE):
        # Must have completion status table
        if '| Component |' not in content and '## Completion Status' not in content:
            errors.append(f"Missing completion grade table: {file_path.name}")

    return errors

def main():
    docs_dir = Path('docs')
    all_errors = []

    for md_file in docs_dir.glob('*.md'):
        all_errors.extend(validate_naming(md_file))
        all_errors.extend(validate_completion_grade(md_file))

    if all_errors:
        print("Document validation failed:")
        for error in all_errors:
            print(f"  - {error}")
        sys.exit(1)

    print("Document validation passed.")
    sys.exit(0)

if __name__ == '__main__':
    main()
```
